// domain/replay/engine.ts
export function replayAt(
  traces: Trace[],
  timestamp: number
): Map<string, Trace> {

  const latest = new Map<string, Trace>()

  for (const t of traces) {
    if (t.ts <= timestamp) {
      const prev = latest.get(t.taskId)
      if (!prev || prev.ts < t.ts) {
        latest.set(t.taskId, t)
      }
    }
  }

  return latest
}

// client

  const snapshot = replayAt(traces, ts)
  render(snapshot)
}

// domain/anomaly/engine.ts
export function detectAnomalies(history: number[]) {
  const anomalies = []

  for (let i = 1; i < history.length; i++) {
    if (Math.abs(history[i] - history[i-1]) > 0.5) {
      anomalies.push({ type: "SPIKE", index: i })
    }
  }

  let flips = 0
  for (let i = 2; i < history.length; i++) {
    const d1 = history[i] - history[i-1]
    const d2 = history[i-1] - history[i-2]
    if (d1 * d2 < 0) flips++
  }

  if (flips > 3) anomalies.push({ type: "OSCILLATION" })

  return anomalies
}
